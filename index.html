~/juna-project/src/main.cpp 
#include 

using namespace juna;

// Modern async server with coroutines
auto create_server() -> Task {
    auto app = App::create();
    
    // Fluent API design
    app.route("/")
       .get([](Context ctx) -> Task {
           co_return json({
               {"framework", "JUNA"},
               {"version", "2.0.0"},
               {"performance", "🚀"}
           });
       });
    
    // Middleware composition
    app.use(middleware::cors())
       .use(middleware::logger())
       .use(middleware::compression());
    
    co_await app.listen(8080);
}

int main() {
    return Runtime::run(create_server());
}
